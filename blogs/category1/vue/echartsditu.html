<template>
  <div class="ech-box">
      <div ref="chartContainer" style="width: 100%; height: 100%;" />
  </div>
</template>
<script>
import echarts from 'echarts'
import chinaJSON from "./china.json"

export default {
  name: 'LeftTop',
  data() {
    return {
      chart: null,
        //弹窗的信息
      data: [
        {
          name: "上海",
          value: ["121.5304871", "31.0856321"],
          id: 4,
          code: "110000Z0004",
          type: "Z",
          date: "2024-01-11",
          city: "上海市"
        },
        {
          name: "老城动物园",
          value: ["113.489356", "36.94083"],
          id: 3,
          code: "110000D0003",
          type: "D",
          date: "2024-01-16",
          city: "山西省",
        },
        {
          name: "新城植物园",
          value: ["114.489356", "37.94083"],
          id: 2,
          code: "110000D0002",
          type: "Z",
          date: "2023-10-21",
          city: "河北省",
        }
      ]
    }
  },
  mounted() {
    this.renderChart()
    window.addEventListener('resize', this.handleResize)
  },
  beforeDestroy() {
    window.removeEventListener('resize', this.handleResize)
  },
  methods: {
    renderChart() {
      const chartContainer = this.$refs.chartContainer;
      echarts.registerMap('china', chinaJSON);
      this.chart = echarts.init(chartContainer);

      const option = {
        // 弹框配置
        tooltip: {
          show: true,
          triggerOn: "none",
          padding: 0,
          enterable: true,
          transitionDuration: 1,
          textStyle: {
            color: "#000",
            decoration: "none"
          },
            //弹框内容的html
          formatter: (params) => {
            const { data } = params;
            const type = data.type;
            let typeName = "";
            if (type === "D") typeName = "动物园";
            if (type === "Z") typeName = "植物园";
            // Add other type checks as needed...

            const tipHtml = `<div style="width: 280px; background: rgba(22, 80, 158, 0.8); border: 1px solid rgba(7, 166, 255, 0.7);">
                                <div style="color: #fff; padding: 10px; font-size: 16px; border-bottom: 2px solid #0c8bdd;">${data.name} (${data.code})</div>
                                <p style="color: #fff;padding: 5px 10px;font-size: 14px;">公司位置：${data.city}</p>
                                <p style="color: #fff;padding: 5px 10px;font-size: 14px;">交付时间：${data.date}</p>
                                <p style="color: #fff;padding: 5px 10px;font-size: 14px;">设施类型：${typeName}</p>
                              </div>`;
            return tipHtml;
          }
        },
        // 地图配置
        geo: {
          map: 'china', // 中国地图
          roam: true, // 鼠标缩放的效果
          // 地图的位置调试
          top: 100,
          zoom: 3,
          bottom: 100,
          // 地图上的文字的设置
          label: {
            show: true, // 文字显示出来
            color: 'white',
            fontSize: 8,
          },
          itemStyle: {
            // 每一个多边形的样式

            color: {
              type: 'linear',
              x: 0,
              y: 0,
              x2: 0,
              y2: 1,
              colorStops: [
                {
                  offset: 0,
                  color: '#073684', // 0% 处的颜色
                },
                {
                  offset: 1,
                  color: '#061E3D', // 100% 处的颜色
                },
              ],
              global: false, // 缺省为 false
            },
            // opacity: 0.8, //透明度
            // shadowBlur: 5,
            // shadowColor: 'rgba(49, 139, 226, 0.5)',
            borderColor: "#00F7FF"

          },
          // 地图高亮的效果
          emphasis: {
            itemStyle: {
              color: '#2075A7',

            },
            label: {
              // fontSize: 16,//hover时候是否文字变大
              color: '#fff',//hover时候文字颜色
            },
          },
        },
        // 系统图表配置项
        series: [
          {
            name: "",
            type: "scatter",
            coordinateSystem: "geo",
            data: this.data,
            symbol: "pin",
            symbolSize: 30,
            label: {
              normal: {
                show: false
              },
              emphasis: {
                show: false
              }
            },
            itemStyle: {
              normal: {
                color: "#33C481"
              },
              emphasis: {
                color: "#f00"
              }
            }
          }
        ]
      };

      this.chart.setOption(option);
      // 鼠标经过效果
      this.chart.on("mouseover", params => {
        if (params.componentType === "series" && params.componentSubType === "scatter") {
          const index = params.dataIndex;
          this.chart.dispatchAction({
            type: "downplay",
            seriesIndex: 0
          });
          this.chart.dispatchAction({
            type: "showTip",
            seriesIndex: 0,
            dataIndex: index
          });
          this.chart.dispatchAction({
            type: "highlight",
            seriesIndex: 0,
            dataIndex: index
          });
        }
      });
      // 鼠标经过其他地图区域时候弹窗消失
      this.chart.getZr().on('mouseover', () => {
        this.hidePopup();
      });
    },
    handleResize() {
      if (this.chart) {
        this.chart.resize()
      }
    },
    hidePopup() {
      this.chart.dispatchAction({
        type: 'hideTip'
      });
    },
  }
}
</script>
